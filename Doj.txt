    // Benutzer-Sync mit Firebase Cloud
    db.ref('users').on('value', (snapshot) => {
        const data = snapshot.val();
        if (data) {
            users = Object.keys(data).map(key => ({ id: key, ...data[key] }));
        } else {
            // Standard User falls DB leer
            users = [{ u: "Admin", p: "DOJ2026", r: "admin" }];
        }
        if(currentUser && currentUser.r === 'admin') renderUserList();
    });

    function handleLogin() {
        const u = document.getElementById('userInput').value;
        const p = document.getElementById('passInput').value;
        const found = users.find(x => x.u === u && x.p === p);

        if(found) {
            currentUser = found;
            document.getElementById('login-overlay').style.display = 'none';
            document.getElementById('mainApp').style.display = 'flex';
            
            const badge = document.getElementById('uBadge');
            badge.innerText = currentUser.u.toUpperCase();
            badge.className = "badge " + (currentUser.r === 'admin' ? 'bg-admin' : 'bg-user');

            if(currentUser.r === 'admin') {
                document.getElementById('admin-panel').style.display = 'flex';
                renderUserList();
            }
            renderLaws(laws);
        } else {
            document.getElementById('err').style.display = 'block';
        }
    }

    function renderLaws(arr) {
        const list = document.getElementById('lawList');
        list.innerHTML = '';
        arr.forEach(l => {
            const d = document.createElement('div');
            d.className = `law-item ${selected.includes(l.id) ? 'selected' : ''}`;
            d.onclick = () => {
                selected.includes(l.id) ? selected = selected.filter(x => x !== l.id) : selected.push(l.id);
                renderLaws(arr); updateUI();
            };
            d.innerHTML = `<div><strong>${l.id} ${l.de}</strong><br><small style="color:#555">${l.en}</small></div>
                           <div style="text-align:right"><span style="color:var(--danger)">${l.j}M</span><br><span style="color:var(--success)">$${l.f}</span></div>`;
            list.appendChild(d);
        });
    }

    function updateUI() {
        const s = document.getElementById('selLaws');
        s.innerHTML = selected.length ? '' : 'Keine Einträge gewählt.';
        selected.forEach(id => {
            const l = laws.find(x => x.id === id);
            s.innerHTML += `<div style="display:flex; justify-content:space-between; margin-bottom:5px; border-bottom:1px solid #222"><span>${l.de}</span><span>${l.j}M / $${l.f}</span></div>`;
        });
        calc();
    }

    function calc() {
        let j = 0, f = 0;
        selected.forEach(id => {
            const l = laws.find(x => x.id === id);
            j += l.j; f += l.f;
        });
        const m = parseFloat(document.getElementById('mod').value);
        document.getElementById('fJail').innerText = Math.min(Math.round(j * m), 120) + " Monate";
        document.getElementById('fFine').innerText = "$" + Math.min(Math.round(f * m), 90000).toLocaleString();
    }

    // ADMIN LOGIC (Cloud-basiert)
    function renderUserList() {
        const l = document.getElementById('userList');
        l.innerHTML = '';
        users.forEach((u, i) => {
            l.innerHTML += `<div class="user-item">
                <span>${u.u} <small>(${u.r})</small></span>
                <span class="delete-ico" onclick="removeUser('${u.id}')">X</span>
            </div>`;
        });
    }

    function addUser() {
        const u = document.getElementById('nU').value;
        const p = document.getElementById('nP').value;
        const r = document.getElementById('nR').value;
        if(u && p) {
            db.ref('users').push({ u, p, r });
            document.getElementById('nU').value = ''; 
            document.getElementById('nP').value = '';
        }
    }

    function removeUser(id) {
        if(!id) return alert("Standard-Admin kann nur in der Datenbank gelöscht werden.");
        db.ref('users/' + id).remove();
    }

    function filterLaws() {
        const q = document.getElementById('search').value.toLowerCase();
        renderLaws(laws.filter(l => l.de.toLowerCase().includes(q) || l.id.includes(q)));
    }

    function reset() { selected = []; updateUI(); renderLaws(laws); }
